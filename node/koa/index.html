export const CustomRouting = async url => {
  let html = '<h1>404</h1>';
  switch (url) {
    case '/':
      html = '<h1>Home Page</h1>';
      break;
    case '/about':
      html = '<h1>About Page</h1>';
      break;
    default:
      break;
  }
  return html;
};

      # 获取参数

      # GET

      TIP

      获取 GET 方法的参数有两种方式

      • 从上下文中直接获取

        • 对象: ctx.query
        • 字符串: ctx.querystring

      • 上下文的 request对象中获取

        • 对象: ctx.request.query
        • 字符串: ctx.request.querystring
